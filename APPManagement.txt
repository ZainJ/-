# -*- coding: utf8 -*-
import LoginCase
from selenium import webdriver
import time
browser1 = webdriver.Firefox()
print '窗口最大化'
browser1.maximize_window()
# 登录业务后台
LoginCase.Login(browser1)
time.sleep(3)
# 点击APP应用管理业务
browser1.find_element_by_class_name('icon-mobile-phone').click()
time.sleep(3)
# 验证是否成功跳转到APP应用管理页面
currUrl = browser1.current_url
if currUrl == 'http://172.16.8.231:12123/sys/m/home/parentid?syjcd=1010':
    print 'APP应用管理界面打开成功'
else:
    print 'APP应用管理界面打开失败'
# 点击待审的消息推送的数字
browser1.find_element_by_xpath('//div[@class="admin-block span6"]/ul/li').click()
# 验证是否成功跳转到APP消息推送审核页面
if currUrl == 'http://172.16.8.231:12123/sys/m/home/parentid?syjcd=1010':
    print '成功跳转到APP消息推送审核页面'
else:
    print 'APP消息推送审核页面跳转失败'
